FIND THE SMALLEST ELEMENT IN THE INTEGER ARRAY USING RECURSION.

        #include <stdio.h>
      
    void getArray(int arr[],int s)
    {
        int i,j;
        printf("Enter Value of Array\n");
        for ( i = 0; i < s; i++)
        {
            scanf("%d",&arr[i]);
        }
        
    }
    int findmin(int arr[], int n, int min) {
        if (n == 0) {
            return min;
        }
    
        if (arr[n - 1] < min) {
            min = arr[n - 1];
        }
    
        return findmin(arr, n - 1, min);
    }
    
    int main() {
        int arr[10] ,s=10;
        getArray(arr,s);    
        int smallest = findmin(arr, s, arr[0]);
        printf("The smallest element in the array is: %d\n", smallest);
    
        return 0;
    }
