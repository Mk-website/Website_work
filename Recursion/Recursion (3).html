CALCULATE THE SUM OF NATURAL NUMBERS TILL 10 USING RECURSION.

        #include <stdio.h>

            int sum(int num)
            {
                if (num==0)
                {
                    return 0;
                }
                
                    return num+sum(num-1); 
            }
            void main()
            {
                int num =10;
                printf("SUM OF 10 NATURAL NUMBERS %d",sum(num));
            }
