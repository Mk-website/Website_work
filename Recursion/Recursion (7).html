ENTER THE NUMBER FROM THE USER AND ADD ALL DIGITS OF THE NUMBER USING RECURSION.

        #include <stdio.h>

            int sumOfDigits(int num) {
                if (num == 0) {
                    return 0;
                } else {
                    return num % 10 + sumOfDigits(num / 10);
                }
            }
            
            int main() {
                int number;
                
                printf("Enter a number: ");
                scanf("%d", &number);
            
                int sum = sumOfDigits(number);
            
                printf("The sum of the digits of %d is %d\n", number, sum);
            
                return 0;
            }

OUTPUT

        Enter a number: 2563
        The sum of the digits of 2563 is 16
