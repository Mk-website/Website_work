CALCULATE FACTORIAL OF A NUMBER USING RECURSION

        #include <stdio.h>
      
            int factorial(int num)
            {
                if (num==0 || num ==1)
                {
                    return 1;
                }
                else
                return num*factorial(num-1);
            }
            void main()
            {
                int num;
                printf("Enter Number\n");
                scanf("%d",&num);
                printf("Factorial of %d is %d",num,factorial(num));
            }
