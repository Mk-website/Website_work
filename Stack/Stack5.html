ENTER THE STRING FROM THE USER AND REVERSE THE STRING USING STACKS.

    #include <stdio.h>
    int push(char stack[], int i, char op, int s);
    int pop(char stack[], int i);
    int peek(char stack[], int i);
    
    void get(char str[], int s) {
        int i;
        char c;
        for (i = 0; i < s; i++) {
            c = getchar();
            if (c == '\n') {
                break;
            }
            str[i] = c;
        }
        str[i] = '\0';
    }
    
    void reverseString(char str[], char dest[]) {
        char rev[100];
        int i = -1, j;
        for (j = 0; str[j] != '\0'; j++) {
            i = push(rev, i, str[j], 100);
        }
        for (j = 0; str[j] != '\0'; j++) {
            dest[j] = rev[peek(rev, i)];
            i = pop(rev, i);
        }
        dest[j] = '\0';
       
    }
    
    int main() {
        char stack[100], str[100];
        printf("Enter String\n");
        get(str, 100);
        reverseString(str, stack);
        printf("Reverse string is %s\n", stack);
        return 0;
    }
    
    int push(char stack[], int i, char op, int s) {
        if (i == s - 1) {
            printf("Stack is full\n");
            return -1;
        } else {
            i++;
            stack[i] = op;
            return i;
        }
    }
    
    int pop(char stack[], int i) {
        if (i == -1) {
            printf("Stack is empty\n");
            return -1;
        } else {
            i--;
            return i;
        }
    }
    
    int peek(char stack[], int i) {
        if (i == -1)
            return -1;
        else
            return i;
    }
