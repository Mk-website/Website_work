CREATE A STACK IMPLEMENTING THE OPERATIONS ON AN INTEGER ARRAY (PUSH, POP, PEEK, ISEMPTY, ISFULL)

#include <stdio.h>
      
int push(int[],int,int,int);
int pop(int[],int,int,int);
int peek(int[],int);
void main()
{
    int arr[10],i=-1,s=10,ch,val;
    do
    {
        printf("Enter your choice\n1.Push\n2.pop\n3.peek\n4.Exit\n");
        scanf("%d",&ch);
        switch (ch)
        {
        case 1:
         printf("Enter Value\n");
         scanf("%d",&val);
         i=push(arr,s,i,val);
            break;
        case 2:
        i=pop(arr,s,i,val);
        if(i!=-1)
        printf("Value poped\n");
        else
        printf("stack is Empty\n");
        break;
        case 3:
        i=peek(arr,i);
        if(i==-1)
        printf("Stack is empty\n");
        else
        printf("topest value is %d\n",arr[i]);
        break;
        case 4:
        printf("Thanks\n");
        default:
            break;
        }
    } while (ch!=4);
    
}
int push(int stack[],int s,int i,int val)
{
    if(i>=s)
    {
        return -1;
    }
    i++;
    stack[i]=val;
    return i;
}
int pop(int stack[],int s,int i,int val)
{
    if(i==-1)
    return -1;
    else
    i--;
    val=stack[i];
    return i;
}
int peek(int stack[],int i)
{
    if(i==-1)
    return -1;
    else
    return i;
}
