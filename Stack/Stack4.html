#include <stdio.h>
#include<stdlib.h>
struct stack
{
    char data;
    struct stack *next;
};
typedef struct stack node;

int isEmpty(node *top)
{
    if (top==NULL)
    {
        return 1;
    }
    return 0;
}
int isFull(node *ptr)
{
    if (ptr == NULL)
    {
        return 1;
    }
    return 0;
}
node* push(node *start,char val)
{
    node *nn;
    nn=(node*)malloc(sizeof(node));
    if (isFull(nn))
    {
       return NULL;
    }
     nn->data=val;
     nn->next=start;
        start=nn;
        return start;
}
node* pop(node *start)
{
    node *temp;
    if (isEmpty(start))
    {
        printf("Stack is Empty\n");
        return 0;
    }
    temp=start;
    start=temp->next;
    free(temp);
    return start;
    
}
int peek(node *start)
{
    if (isEmpty(start))
    {
        return 0;
    }
    
    return 1;
}
