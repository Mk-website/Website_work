#include<stdio.h>
#include<conio.h>
#include<stdlib.h>
    
    struct Stack
    {
        int data;
        struct Stack *next;
    };
    typedef struct Stack node;
    node* push(node *start,int key)
    {
        struct Stack *temp;
        if(start==NULL)
        {
            temp=(struct Stack *)malloc(sizeof(struct Stack));
            temp->data=key;
            temp->next=NULL;
            start=temp;
            return start;
        }
        else
        {
             temp=(struct Stack *)malloc(sizeof(struct Stack));
            temp->data=key;
            temp->next=start;
            start=temp;
            return start;
        }
    }
    node* pop(node *start)
    {
        struct Stack *temp;
        if (start==NULL)
        {
            printf("Stack is empty\n");
        }
         else{
            temp=start;
            printf("\n%d is popped ",temp->data);
            temp=start->next;
            free(start);
            start=temp;
            return start;
         }
    }
    void peek(node *start)
    {
      if (start==NULL)
        {
            printf("\nStack is empty\n");
        }
      else  
        printf("%d is topest value\n",start->data);
    }
    void main()
    {
        int ch,val;
        node *start=NULL;
         do

            printf("\nEnter your choice\n1.Push\n2.pop\n3.peek\n4.Exit\n");

            start =push(start,val);
                break;

            case 4:
