INSERTION SORT

Program 5

ENTER A INTEGER ARRAY FROM THE USER AND PERFORM INSERTION SORT ALGORITHM THROUGH ITERATION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

#include <stdio.h>
#include <time.h>

void insertionSort(int arr[], int n) {
    int comparisons = 0;
    int swaps = 0;

    for (int i = 1; i < n; i++) {
        int key = arr[i];
        int j = i - 1;

        while (j >= 0 && arr[j] > key) {
            comparisons++;
            arr[j + 1] = arr[j];
            j--;
            swaps++;
        }
        arr[j + 1] = key;
    }

    printf("Sorted Array: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    printf("Comparisons: %d\n", comparisons);
    printf("Swaps: %d\n", swaps);
}

int main() {
    int n=10, arr[10];
    printf("Enter %d integers : ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    clock_t start_time, end_time;
    double cpu_time_used;

    start_time = clock();
    insertionSort(arr, n);
    end_time = clock();

    cpu_time_used = ((double)(end_time - start_time)) / CLOCKS_PER_SEC;

    // Calculate worst-case and best-case time complexity
    int worst_case_comparisons = (n * (n - 1)) / 2;
    int best_case_comparisons = n - 1;

    printf("Execution Time (seconds): %f\n", cpu_time_used);
    printf("Worst-case Time Complexity: %d comparisons\n", worst_case_comparisons);
    printf("Best-case Time Complexity: %d comparisons\n", best_case_comparisons);

    return 0;
}
