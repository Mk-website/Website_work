INSERTION SORT

Program 6

ENTER A INTEGER ARRAY FROM THE USER AND PERFORM INSERTION SORT ALGORITHM THROUGH RECURSION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

#include <stdio.h>
#include <time.h>

void insertionSortRecursive(int arr[], int n) {
    if (n <= 1) {
        return;
    }

    insertionSortRecursive(arr, n - 1);

    int key = arr[n - 1];
    int j = n - 2;

    while (j >= 0 && arr[j] > key) {
        arr[j + 1] = arr[j];
        j--;
    }

    arr[j + 1] = key;
}

int main() {
    int n=10, arr[10];
    printf("Enter %d integers : ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    clock_t start_time, end_time;
    double cpu_time_used;

    start_time = clock();
    insertionSortRecursive(arr, n);
    end_time = clock();

    cpu_time_used = ((double)(end_time - start_time)) / CLOCKS_PER_SEC;

    printf("Sorted Array: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    // Calculate worst-case and best-case time complexity
    int worst_case_comparisons = (n * (n - 1)) / 2;
    int best_case_comparisons = n - 1;

    printf("Execution Time (seconds): %f\n", cpu_time_used);
    printf("Worst-case Time Complexity: %d comparisons\n", worst_case_comparisons);
    printf("Best-case Time Complexity: %d comparisons\n", best_case_comparisons);

    return 0;
}
