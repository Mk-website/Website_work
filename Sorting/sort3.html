BUBBLE SORT

Program 3

ENTER A INTEGER ARRAY FROM THE USER AND PERFORM BUBBLE SORT ALGORITHM THROUGH ITERATION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

        #include <stdio.h>
#include <time.h>

void swap(int *x, int *y) {
    int temp = *x;
    *x = *y;
    *y = temp;
}

void bubbleSort(int arr[], int n) {
    int comparisons = 0;
    int swaps = 0;
    int sorted = 0;

    for (int i = 0; i < n - 1; i++) {
        sorted = 1;
        for (int j = 0; j < n - 1 - i; j++) {
            comparisons++;
            if (arr[j] > arr[j + 1]) {
                swap(&arr[j], &arr[j + 1]);
                swaps++;
                sorted = 0;
            }
        }
        if (sorted == 1) {
            break;
        }
    }

    printf("Sorted Array: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    printf("Comparisons: %d\n", comparisons);
    printf("Swaps: %d\n", swaps);
}

int main() {
   int n=10, arr[10];
    printf("Enter %d integers separated by spaces: ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    clock_t start_time, end_time;
    double cpu_time_used;

    start_time = clock();
    bubbleSort(arr, n);
    end_time = clock();

    cpu_time_used = ((double)(end_time - start_time)) / CLOCKS_PER_SEC;

    // Calculate worst-case and best-case time complexity
    int worst_case_comparisons = (n * (n - 1)) / 2;
    int best_case_comparisons = (n - 1);

    printf("Execution Time (seconds): %f\n", cpu_time_used);
    printf("Worst-case Time Complexity: %d comparisons\n", worst_case_comparisons);
    printf("Best-case Time Complexity: %d comparisons\n", best_case_comparisons);

    return 0;
}
