ENTER A INTEGER ARRAY FROM THE USER AND PERFORM MERGE SORT ALGORITHM THROUGH ITERATION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

        #include <stdio.h>
        
        #include <time.h>
        
        void merge(int arr[], int left, int mid, int right) {
            int i, j, k;
            int n1 = mid - left + 1;
            int n2 = right - mid;
        
            int L[n1], R[n2];
        
            for (i = 0; i < n1; i++)
                L[i] = arr[left + i];
            for (j = 0; j < n2; j++)
                R[j] = arr[mid + 1 + j];
        
            i = 0;
            j = 0;
            k = left;
        
            while (i < n1 && j < n2) {
                if (L[i] <= R[j]) {
                    arr[k] = L[i];
                    i++;
                } else {
                    arr[k] = R[j];
                    j++;
                }
                k++;
            }
        
            while (i < n1) {
                arr[k] = L[i];
                i++;
                k++;
            }
        
            while (j < n2) {
                arr[k] = R[j];
                j++;
                k++;
            }
        }
        
        void mergeSort(int arr[], int n) {
            int curr_size;
            int left_start;
        
            for (curr_size = 1; curr_size <= n - 1; curr_size = 2 * curr_size) {
                for (left_start = 0; left_start < n - 1; left_start += 2 * curr_size) {
                    int mid = left_start + curr_size - 1;
                    int right_end = (left_start + 2 * curr_size - 1 < n - 1) ? (left_start + 2 * curr_size - 1) : (n - 1);
                    merge(arr, left_start, mid, right_end);
                }
            }
        }
        
        int main() {
            int n=10, arr[10];
            printf("Enter %d integers : ", n);
            for (int i = 0; i < n; i++) {
                scanf("%d", &arr[i]);
            }
        
            int worst_case[n];
            int best_case[n];
        
            // Copy the original array for best-case scenario
            for (int i = 0; i < n; i++) {
                worst_case[i] = arr[i];
                best_case[i] = arr[i];
            }
        
            clock_t start_time, end_time;
        
            // Worst-case time complexity
            start_time = clock();
            mergeSort(worst_case, n);
            end_time = clock();
            printf("Sorted array in worst case:\n");
            for (int i = 0; i < n; i++) {
                printf("%d ", worst_case[i]);
            }
            printf("\nWorst case time taken: %.6f seconds\n", (double)(end_time - start_time) / CLOCKS_PER_SEC);
        
            // Best-case time complexity (already sorted array)
            start_time = clock();
            mergeSort(best_case, n);
            end_time = clock();
            printf("Sorted array in best case:\n");
            for (int i = 0; i < n; i++) {
                printf("%d ", best_case[i]);
            }
            printf("\nBest case time taken: %.6f seconds\n", (double)(end_time - start_time) / CLOCKS_PER_SEC);
        
            return 0;
        }
