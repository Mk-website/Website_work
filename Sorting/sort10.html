QUICK SORT

Program 10

ENTER A INTEGER ARRAY FROM THE USER AND PERFORM QUICK SORT ALGORITHM THROUGH RECURSION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

        #include <stdio.h>
#include <time.h>

void swap(int* a, int* b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int partition(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = (low - 1);

    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(&arr[i], &arr[j]);
        }
    }

    swap(&arr[i + 1], &arr[high]);
    return (i + 1);
}

void quicksort(int arr[], int low, int high) {
    if (low < high) {
        int pivot_index = partition(arr, low, high);
        quicksort(arr, low, pivot_index - 1);
        quicksort(arr, pivot_index + 1, high);
    }
}

int main() {
     int n=10, arr[10];
    printf("Enter %d integers : ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int worst_case[n];
    int best_case[n];

    // Copy the original array for best-case scenario
    for (int i = 0; i < n; i++) {
        worst_case[i] = arr[i];
        best_case[i] = arr[i];
    }

    clock_t start_time, end_time;

    // Worst-case time complexity
    start_time = clock();
    quicksort(worst_case, 0, n - 1);
    end_time = clock();
    printf("Sorted array in worst case:\n");
    for (int i = 0; i < n; i++) {
        printf("%d ", worst_case[i]);
    }
    printf("\nWorst case time taken: %.6f seconds\n", (double)(end_time - start_time) / CLOCKS_PER_SEC);

    // Best-case time complexity (already sorted array)
    start_time = clock();
    quicksort(best_case, 0, n - 1);
    end_time = clock();
    printf("Sorted array in best case:\n");
    for (int i = 0; i < n; i++) {
        printf("%d ", best_case[i]);
    }
    printf("\nBest case time taken: %.6f seconds\n", (double)(end_time - start_time) / CLOCKS_PER_SEC);

    return 0;
}
