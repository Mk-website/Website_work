INSERTION SORT

Program 15

ENTER A STRING ARRAY FROM THE USER AND PERFORM INSERTION SORT ALGORITHM THROUGH ITERATION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <time.h>

void insertionSort(char* arr[], int n) {
    for (int i = 1; i < n; i++) {
        char* key = arr[i];
        int j = i - 1;

        while (j >= 0 && strcmp(arr[j], key) > 0) {
            arr[j + 1] = arr[j];
            j--;
        }

        arr[j + 1] = key;
    }
}

int main() {
   int n=10;
    char* arr[n];
    printf("Enter the strings: \n");
    for (int i = 0; i < n; i++) {
        arr[i] = (char*)malloc(100 * sizeof(char));
        scanf("%s", arr[i]);
    }

    char* worst_case[n];
    char* best_case[n];

    // Copy the original array for best-case scenario
    for (int i = 0; i < n; i++) {
        worst_case[i] = strdup(arr[i]);
        best_case[i] = strdup(arr[i]);
    }

    clock_t start_time, end_time;

    // Worst-case time complexity
    start_time = clock();
    insertionSort(worst_case, n);
    end_time = clock();
    printf("Sorted array in worst case:\n");
    for (int i = 0; i < n; i++) {
        printf("%s\n", worst_case[i]);
    }
    printf("Worst case time taken: %.6f seconds\n", (double)(end_time - start_time) / CLOCKS_PER_SEC);

    // Best-case time complexity (already sorted array)
    start_time = clock();
    insertionSort(best_case, n);
    end_time = clock();
    printf("Sorted array in best case:\n");
    for (int i = 0; i < n; i++) {
        printf("%s\n", best_case[i]);
    }
    printf("Best case time taken: %.6f seconds\n", (double)(end_time - start_time) / CLOCKS_PER_SEC);

    // Free dynamically allocated memory
    for (int i = 0; i < n; i++) {
        free(arr[i]);
        free(worst_case[i]);
        free(best_case[i]);
    }

    return 0;
}
