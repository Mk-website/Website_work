Enter a Matrix from user, copy it in another matrix of same size?

#include <stdio.h>
void display(int matrix[3][3])
{
    int i,j;
    for(i=0; i < 3;i++)
    {
        for(j=0; j < 3;j++)
        {
            printf("%d ",matrix[i][j]);
        }
        printf("\n");
    }
}
void copyMatrix(int matrix[3][3],int matrix2[3][3])
{
    int i,j,temp;
    for(i=0; i < 3;i++)
    {
        for(j=0; j < 3;j++)
        {
            
            matrix2[i][j]=matrix[i][j];
        }
        
    }
}
void main()
{
    int matrix[3][3],matrix2[3][3];
    for( i = 0; i < 3; i++)

        {
            printf("Enter Value at A[%d,%d]\t", i+1, j+1);
            scanf("%d", &matrix[i][j]);
        }
        }
    display(matrix);
    copyMatrix(matrix,matrix2);
    printf("Copied matrix\n");
    display(matrix2);
}
