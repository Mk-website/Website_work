Create a tic tac toe game using 2D Arrays

#include <stdio.h>
// Function to initialize the game board
void initialize(char board[3][3]) {
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            board[i][j] = ' ';
        }
    }
}

// Function to display the game board
void display_board(char board[3][3]) {
    printf("  1 2 3\n");
    for (int i = 0; i < 3; i++) {
        printf("%d ", i + 1);
        for (int j = 0; j < 3; j++) {
            printf("%c", board[i][j]);
            if (j < 2) {
                printf("|");
            }
        }
        printf("\n");
        if (i < 2) {
            printf("  -+-+-\n");
        }
    }
}

// Function to check if a move is valid
int is_valid_move(char board[3][3], int row, int col) {
    return (row >= 0 && row < 3 && col >= 0 && col < 3 && board[row][col] == ' ');
}

// Function to check if the game is over
int is_game_over(char board[3][3]) {
    for (int i = 0; i < 3; i++) {
        if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][0] != ' ') {
            return 1;
        }
        if (board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[0][i] != ' ') {
            return 1;
        }
    }
    if (board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[0][0] != ' ') {
        return 1;
    }
    if (board[0][2] == board[1][1] && board[1][1] == board[2][0] && board[0][2] != ' ') {
        return 1;
    }

    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            if (board[i][j] == ' ') {
                return 0; // Game is not over yet
            }
        }
    }
    return -1; // It's a draw
}

int main() {
    char board[3][3];
    char player = 'X';
    int row, col, move;

    initialize(board);

    while (1) {
        display_board(board);

        // Switch players
        if (player == 'X') {
            printf("Player X's turn. Enter row and column (1-3): ");
        } else {
            printf("Player O's turn. Enter row and column (1-3): ");
        }

        // Get the player's move
        move = scanf("%d %d", &row, &col);

        if (move != 2 || !is_valid_move(board, row - 1, col - 1)) {
            printf("Invalid move. Try again.\n");
            while (getchar() != '\n');
            continue;
        }

        board[row - 1][col - 1] = player;

        int game_result = is_game_over(board);

        if (game_result == 1) {
            display_board(board);
            printf("Player %c wins!\n", player);
            break;
        } else if (game_result == -1) {
            display_board(board);
            printf("It's a draw!\n");
            break;
        }

        player = (player == 'X') ? 'O' : 'X';
    }

    return 0;
}
