#include <stdio.h>

    void display(int arr[3][3])
    {
        int i,j,sum[3][3];
        for(i=0;i<3;i++)
        {
            for (j = 0; j < 3; j++)
            {
                printf("%d\t",arr[i][j]);
            }
            printf("\n");
        }
    }
    void subtract(int arr1[3][3],int arr2[3][3])
    {
        int i,j,sum[3][3];
        for(i=0;i<3;i++)
        {
            for (j = 0; j < 3; j++)
            {
                sum[i][j]=arr1[i][j]-arr2[i][j];
    
            }
            
        }
        display(sum);
    }
    void main()
    {
        int arr1[3][3],arr2[3][3];

            printf("For Second Matrix\n");
            printf("Enter Value at B[%d,%d]\t", i+1, j+1);
            scanf("%d", &arr2[i][j]);
        }
        }
        subtract(arr1,arr2);
    }
