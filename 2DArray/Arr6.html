Enter the Matrix from user, multiply it with a scalar value and display the answer?

#include <stdio.h>
void display(int matrix[3][3])
{
    int i,j;
    for(i=0; i < 3;i++)
    {
        for(j=0; j < 3;j++)
        {
            printf("%d ",matrix[i][j]);
        }
        printf("\n");
    }
}
void scaler(int matrix[3][3],int multi)
{
    int i,j;
    for(i=0; i < 3;i++)
    {
        for(j=0; j < 3;j++)
        {
            matrix[i][j]=multi*matrix[i][j];
        }
        
    }
    printf("After Scaler Multiply\n");
    display(matrix);
}
void main()
{
    int matrix[3][3],i,j,n;
    for( i = 0; i < 3; i++)
        {
        for (j = 0; j < 3; j++)
        {
            printf("Enter Value at A[%d,%d]\t", i+1, j+1);
            scanf("%d", &matrix[i][j]);
        }
        }
    display(matrix);
    printf("Enter Number\n");
    scanf("%d",&n);
    scaler(matrix,n);
}
