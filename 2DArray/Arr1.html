Enter a matrix of 4 rows and 2 columns from the user display it in matrix format?

#include <stdio.h>
      
void main()
{
int matrix[4][2],i,j;
printf("Enter value in Matrix\n");
for ( i = 0; i < 4; i++)
{
    for ( j = 0; j < 2; j++)
    {
        scanf("%d",&matrix[i][j]);
    }
    
}
printf("Matrix\n");
for ( i = 0; i < 4; i++)
{
    for ( j = 0; j < 2; j++)
    {
        printf("%d ",matrix[i][j]);
    }
    printf("\n");
}

}
