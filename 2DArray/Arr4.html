void multi(int arr1[3][3],int arr2[3][3])
{
    int i,j,k,mul[3][3];
    for(i=0;i<3;i++)
    {
        for(j=0;j<3;j++)
        {
            mul[i][j]=0;
            for(k=0;k<3;k++)
            {
            mul[i][j]+=arr1[i][k]*arr2[k][j];
            }
        }
    }
    display(mul);
}
