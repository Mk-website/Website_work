ENTER A INTEGER ARRAY FROM THE USER, ENTER AN INTEGER VALUE AND PERFORM LINEAR SEARCH ALGORITHM THROUGH RECURSION.

        #include<stdio.h>
    #include <conio.h>
    #include <dos.h>
    int linearSearch(int arr[], int key, int lower, int n) {
        if (lower == n)
            return -1;
        
        if (arr[lower] == key)
            return lower;
    
        return linearSearch(arr, key, lower+1, n);
    }
    void main(){
        int arr[10], i, key, result;
        clrscr();
        for (i = 0; i < 10; i++)
        {
            printf("Enter Value of position %d\n", i+1);
            scanf("%d", &arr[i]);
        }
    
        printf("Enter Value to be searched\n");
        scanf("%d", &key);
        result = linearSearch(arr, key, 0, 10);
    
        if (result == -1) {
            printf("Element %d not found in the array.\n", key);
        } else {
            printf("Element %d found at position %d.\n", key, result+1);
        }    
        delay(5000);
    }
