        #include <stdio.h>
        #include <string.h>
        #include <conio.h>
        #include <dos.h>
        
        int binarySearch(char arr[], int n, char key) {
            int l = 0, u = n-1;
            while(l <= u)
            {
                int mid = l + (u - l) / 2;
                if (arr[mid] == key)
                {
                    return mid;
                }
                else if(arr[mid] > key){
                    u = mid-1;
                }
                else {
                    l = mid+1;
                }
            }
            return -1; 
        }
        
        void main() {
            char arr[20], key;
            int result;
            clrscr();
            printf("Enter any String\n");
            gets(arr);
        
            printf("Enter Character to be searched\n");
            scanf("%c", &key);
            result = binarySearch(arr, strlen(arr), key);
        
            if (result == -1) {
                printf("Element %c not found in the array.\n", key);
            } else {
                printf("Element %c found at position %d.\n", key, result+1);
            }
        
            delay(5000);
        }
