ENTER A STRING FROM THE USER, ENTER A CHARACTER AND PERFORM LINEAR SEARCH ALGORITHM THROUGH ITERATION.

        #include <stdio.h>
    #include <conio.h>
    #include <dos.h>
    int linearSearch(char str[], char key) {
        int i;
        for (i = 0; str[i]; i++) {
            if (str[i] == key) {
            return i;
            }
        }
        return -1; 
    }
    
    void main() {
        char str[20], key;
        int result;
        clrscr();
        printf("Enter any String\n");
        gets(str);
    
        printf("Enter charater to be searched\n");
        scanf("%c", &key);
    
        result = linearSearch(str, key);
    
        if (result == -1) {
            printf("Element %c not found in the array.\n", key);
        } else {
            printf("Element %c found at position %d.\n", key, result+1);
        }
    
        delay(5000);
    }
