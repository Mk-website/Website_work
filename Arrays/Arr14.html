    printf("Enter the elements of the array:\n");
    for (int i = 0; i <n; i++) {
    scanf("%d", &arr[i]);
    }
    
    printf("All possible combinations:\n");
    
    // Generate all combinations using nested loops
    for (int i = 0; i <n; i++) 
    {
        for (int j = i; j <n; j++) 
        {
        printf("{ ");
    for (int k = i; k <= j; k++) 
    { 
        printf("%d ", arr[k]);
    }
        printf(" }\n"); 
    } 
} return 0; 
}
