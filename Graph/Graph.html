ENTER THE VERTICES AND EDGES AND REPRESENT IN ADJACENCY MATRIX AND PRINT IT.

#include <stdio.h>

// Maximum number of vertices
#define MAX_VERTICES 10

int main() {
    int vertices, edges;
    
    // Get the number of vertices from the user
    printf("Enter the number of vertices (max %d): ", MAX_VERTICES);
    scanf("%d", &vertices);

    if(vertices > MAX_VERTICES || vertices <= 0) {
        printf("Invalid number of vertices. Exiting...\n");
        return 1;
    }

    // Adjacency matrix to represent the graph
    int adjMatrix[MAX_VERTICES][MAX_VERTICES] = {0};

    // Get the number of edges from the user
    printf("Enter the number of edges: ");
    scanf("%d", &edges);

    // Get edges from the user
    printf("Enter the edges (vertex1 vertex2): \n");
    for (int i = 0; i < edges; ++i) {
        int vertex1, vertex2;
        scanf("%d %d", &vertex1, &vertex2);

        // Update the adjacency matrix
        adjMatrix[vertex1][vertex2] = 1;
        adjMatrix[vertex2][vertex1] = 1;  // Assuming an undirected graph
    }

    // Print the adjacency matrix
    printf("\nAdjacency Matrix:\n");
    for (int i = 1; i <= vertices; ++i) {
        for (int j = 1; j <= vertices; ++j) {
            printf("%d ", adjMatrix[i][j]);
        }
        printf("\n");
    }

    return 0;
}
