        int count(char str[],int s,int i)
        {
            int sum=0;
            for(i=0;i < s;i++)
            {
                if(str[i]=='a' || str[i]=='A' || str[i]=='o' || str[i]=='O' || str[i]=='e' || str[i]=='E' || str[i]=='i' || str[i]=='I' || str[i]=='u' || str[i]=='U' )
                sum++;
            }
            printf("Number of vowel is %d ",sum);
        }
