ENTER THE STRING FROM THE USER AND CONVERT INTO A FLOAT VALUE?

#include <stdio.h>
void get(char str[],int s)
{
    int i;
    char c;
    for ( i = 0; i < s; i++)
    {
        c=getchar();
        if(c=='\n')
        break;
        str[i]=c;
    }
    str[i]='\0';
}
float stringToFloat(const char *str) {
    float result = 0.0f;
    float fraction = 0.1f;
    int sign = 1;

    // Check for a negative sign
    if (*str == '-') {
        sign = -1;
        str++;
    }

    // Process the integer part of the number
    while (*str != '\0' && *str >= '0' && *str <= '9') {
        result = result * 10.0f + (*str - '0');
        str++;
    }

    // Process the fractional part of the number
    if (*str == '.') {
        str++;
        while (*str != '\0' && *str >= '0' && *str <= '9') {
            result += (*str - '0') * fraction;
            fraction /= 10.0f;
            str++;
        }
    }

    return result * sign;
}

int main() {
    char str[50];
    printf("Enter String\n");
    get(str,50);
     // Your input string
    float num = stringToFloat(str);

    printf("String as float: %f\n", num);
    return 0;
}
