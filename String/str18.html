ENTER THE STRING FROM THE USER AND CONVERT INTO AN INTEGER VALUE?

#include <stdio.h>
      
void get(char str[],int s)
{
    int i;
    char c;
    for ( i = 0; i < s; i++)
    {
        c=getchar();
        if(c=='\n')
        break;
        str[i]=c;
    }
    str[i]='\0';
}
int stringToInteger(const char *str) {
    int result = 0;
    int sign = 1; // to handle negative numbers

    // Check for a negative sign
    if (*str == '-') {
        sign = -1;
        str++; // Move past the negative sign
    }

    // Process each character in the string
    while (*str != '\0') {
        if (*str >= '0' && *str <= '9') {
            result = result * 10 + (*str - '0');
        } else {
            // If a non-digit character is encountered, stop parsing
            break;
        }
        str++;
    }

    return result * sign;
}

int main() {
    char str[50];
    printf("Enter String \n");
    get(str,50);
    int num = stringToInteger(str);

    printf("String as integer: %d\n", num);
    return 0;
}
