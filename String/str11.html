void reverse(char str[],int l,char str2[])
{
    int j=0;
    
    for (int i = l-1; i>=0; i--)
    {
        str2[j]=str[i];
        j++;
    }
    str2[j]='\0';
    display(str2);
}
int palindrome(char str[],char str2[],int p,int l)
{
    int i;
    for ( i = 0; i <=l; i++)
    {
        if(str[i]!=str2[i])
        {
            p=1;
            break;
        }
    }
    if(p==0)
    printf("\nPalindrome String\n");
    else
    printf("\nNot Palindrome");
}
