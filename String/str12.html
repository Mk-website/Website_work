        #include<stdio.h>
#include<conio.h>
void get(char str[],int s)
{
    int i;
    char c;
    printf("Enter Sentance\n");
    for ( i = 0; i < s; i++)
    {
        c=getchar();
        if(c=='\n')
        break;
        str[i]=c;
    }
    str[i]='\0';
}
void display(char str[])
{
    for ( int i = 0; str[i]!='\0'; i++)
    {
        printf("%c\n",str[i]);
    }
    
}
void main()
{
    char str[50];
    get(str,50);
    display(str);
}
