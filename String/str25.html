        printf("Before Trim\n");
        printf("%c", '(');
        printf("%s", str);

        for (i = 0; str[i]; i++)
        {

            break;

            i = i+j;
    
        }
        for(i = 0, j = 0; str[i]; i++){

           str[j++] = str[i];
        }
