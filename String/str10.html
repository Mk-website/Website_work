ENTER THE STRING FROM USER, ENTER THE INDEX NUMBER AND PRINT CHARACTER PRESENT ON THAT INDEX ?

#include <stdio.h>
void get(char str[],int s)
{
    int i;
    char c;
    for ( i = 0; i < s; i++)
    {
        c=getchar();
        if (c=='\n')
        {
            break;
        }
        str[i]=c;
        
    }
    str[i]='\0';
}

void main()
{
    char str[50];
    int i;
    get(str,50);
    printf("Enter Index\n");
    scanf("%d",&i);
    printf(" %c",str[i]);
}
