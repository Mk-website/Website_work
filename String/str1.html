#include<stdio.h>
void get(char str[],int s)
{
int i;
char c;
for ( i = 0; i < s; i++)
{
    c=getchar();
    if (c=='\n')
    {
        break;
    }
    str[i]=c;  
}
str[i]='\0';
}
void main()
{
    char str[50];
    printf("Enter String: \n");
    get(str,50);
    for (int i = 0; str[i]!='\0'; i++)
    {
        if (str[i]<='a' && str[i]>='z')
        {
           str[i]=str[i]-32;
        }
        
    }
    printf("UpperCase %s",str);
}
