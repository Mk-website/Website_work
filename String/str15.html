        #include <stdio.h>
      
        void getstr(char str[],int s)
        {
            int i;
            char c;
            for ( i = 0; i < s; i++)
            {
                c=getchar();
                if (c=='\n')
                {
                    break;
                }
                str[i]=c;
                
            }
            str[i]='\0';
            
        }
        int main() {
            char str[50];
            getstr(str,50);
            char characterToFind;
            
            printf("Enter a character to check if it exists in the string: ");
            scanf("%c", &characterToFind);
        
            int found = 0;
        
            for (int i = 0; str[i] != '\0'; i++) {
                if (str[i] == characterToFind) {
                    found = 1;
                    break; // Exit the loop if the character is found
                }
            }
        
            if (found) {
                printf("The character '%c' exists in the string.\n", characterToFind);
            } else {
                printf("The character '%c' does not exist in the string.\n", characterToFind);
            }
        
            return 0;
        }
