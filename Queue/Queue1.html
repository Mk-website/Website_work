#include <stdio.h>
int enqueue(int[],int,int,int);
int del(int,int);
int peek(int,int);

int isEmpty(int front, int rear)
{
    if (front == rear)
    {
        return 1;
    }
    return 0;
}

int isFull(int rear, int size)
{
    if (rear == size)
    {
        return 1;
    }
    return 0;
}

{

        printf("Enter Value\n");
        scanf("%d",&val);
        check=enqueue(queue,f,r,val);
          if (check == -1)

                printf("%d inserted to Queue\n", val);
                r = check;

        break;

                f = check;

}
